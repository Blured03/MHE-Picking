เชื่อมต่อ Google Sheets

ยังไม่ได้เชื่อมต่อ — ข้อมูลจะถูกเก็บในเครื่องนี้เท่านั้น

วิธีตั้งค่า (ทำครั้งเดียว)

  1. เปิด Google Sheets สร้างไฟล์ใหม่ 1 ไฟล์ (ตั้งชื่อเช่น "MHE Picking Data")
  2. ไปที่เมนู ส่วนขยาย (Extensions) → Apps Script
  3. ลบโค้ดเดิมทั้งหมด แล้ววางโค้ดจากกล่องด้านล่างแทน
  4. กด บันทึก (ไอคอนแผ่นดิสก์) แล้วกด ปรับใช้ (Deploy) → ปรับใช้ใหม่ (New deployment)
  5. เลือกประเภทเป็น เว็บแอป (Web app) — ตั้งค่า "Execute as" = Me และ "Who has access" = Anyone
  6. กด ปรับใช้ (Deploy) แล้วอนุญาตสิทธิ์ (Authorize access) ตามที่ระบบขอ
  7. คัดลอก URL ของเว็บแอปที่ได้ มาวางในช่องด้านบน แล้วกด "บันทึก URL" และ "ทดสอบการเชื่อมต่อ"

หากแก้โค้ดภายหลัง ต้องไปที่ Deploy → Manage deployments → กดไอคอนดินสอ → เลือก Version เป็น "New version" → Deploy ใหม่ทุกครั้ง

โค้ด Google Apps Script

// วางโค้ดนี้ทั้งหมดแทนที่ Code.gs
const SUMMARY_SHEET = 'สรุปรายวัน';
const DETAIL_SHEET  = 'รายละเอียดเครื่อง';
const NOTE_SHEET    = 'หมายเหตุประจำวัน';

function doPost(e){
  const data = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  saveDay(ss, data);
  return json_({status:'ok'});
}

function doGet(e){
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const action = (e.parameter.action || 'ping');
  if(action === 'history') return json_(getAllHistory(ss));
  if(action === 'day') return json_(getDay(ss, e.parameter.date));
  return json_({status:'ready'});
}

function json_(obj){
  return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}

function getOrCreateSheet_(ss, name, headers){
  let sh = ss.getSheetByName(name);
  if(!sh){ sh = ss.insertSheet(name); sh.appendRow(headers); }
  return sh;
}

function fmtDate_(d){
  if(Object.prototype.toString.call(d) === '[object Date]'){
    return Utilities.formatDate(d, Session.getScriptTimeZone(), 'yyyy-MM-dd');
  }
  return String(d);
}

function removeRowsForDate_(sh, date){
  const values = sh.getDataRange().getValues();
  for(let i = values.length - 1; i >= 1; i--){
    if(fmtDate_(values[i][0]) === date) sh.deleteRow(i + 1);
  }
}

function brokenList_(units){
  return units.filter(u => u.status === 'repair')
    .map(u => u.number + (u.remark ? ' (' + u.remark + ')' : ''))
    .join(', ');
}

function computeSummaryRows_(data){
  const rows = [];
  (data.categories || []).forEach(cat => {
    if(cat.kind === 'unit'){
      const units = cat.units || [];
      const subtypes = cat.subtypes || [];
      if(subtypes.length){
        subtypes.forEach(st => {
          const grp = units.filter(u => u.subtype === st);
          const repair = grp.filter(u => u.status === 'repair').length;
          rows.push({label: cat.name + ' ' + st, total: grp.length, repair, ready: grp.length - repair, remark: brokenList_(grp)});
        });
        const orphans = units.filter(u => subtypes.indexOf(u.subtype) === -1);
        if(orphans.length){
          const repair = orphans.filter(u => u.status === 'repair').length;
          rows.push({label: cat.name + ' อื่นๆ', total: orphans.length, repair, ready: orphans.length - repair, remark: brokenList_(orphans)});
        }
      } else {
        const repair = units.filter(u => u.status === 'repair').length;
        rows.push({label: cat.name, total: units.length, repair, ready: units.length - repair, remark: brokenList_(units)});
      }
    } else {
      const total = Number(cat.total) || 0, repair = Number(cat.repair) || 0;
      rows.push({label: cat.name, total, repair, ready: total - repair, remark: cat.remark || ''});
    }
  });
  return rows;
}

function saveDay(ss, data){
  const date = data.date;

  const detail = getOrCreateSheet_(ss, DETAIL_SHEET, ['Date','หมวดหมู่','เบอร์','ชนิดย่อย','สถานะ','หมายเหตุ']);
  removeRowsForDate_(detail, date);
  (data.categories || []).forEach(cat => {
    if(cat.kind === 'unit'){
      (cat.units || []).forEach(u => {
        detail.appendRow([date, cat.name, u.number, u.subtype || '-', u.status === 'ready' ? 'พร้อมใช้' : 'ซ่อม', u.remark || '']);
      });
    }
  });

  const summary = getOrCreateSheet_(ss, SUMMARY_SHEET, ['Date','Category','ทั้งหมด','พร้อมใช้','ซ่อม','หมายเหตุ']);
  removeRowsForDate_(summary, date);
  const rows = computeSummaryRows_(data);
  rows.forEach(r => summary.appendRow([date, r.label, r.total, r.ready, r.repair, r.remark]));
  const totals = rows.reduce((a, r) => ({total: a.total + r.total, ready: a.ready + r.ready, repair: a.repair + r.repair}), {total:0, ready:0, repair:0});
  summary.appendRow([date, 'รวมทั้งหมด', totals.total, totals.ready, totals.repair, '']);

  const noteSh = getOrCreateSheet_(ss, NOTE_SHEET, ['Date','หมายเหตุ','บันทึกเมื่อ']);
  removeRowsForDate_(noteSh, date);
  noteSh.appendRow([date, data.note || '', new Date()]);
}

function getAllHistory(ss){
  const summary = ss.getSheetByName(SUMMARY_SHEET);
  if(!summary) return {};
  const values = summary.getDataRange().getValues();
  const map = {};
  for(let i = 1; i < values.length; i++){
    const [d, category, total, ready, repair] = values[i];
    const date = fmtDate_(d);
    if(category === 'รวมทั้งหมด') continue;
    if(!map[date]) map[date] = {total:0, ready:0, repair:0};
    map[date].total += Number(total) || 0;
    map[date].ready += Number(ready) || 0;
    map[date].repair += Number(repair) || 0;
  }
  return map;
}

function getDay(ss, date){
  const detail = ss.getSheetByName(DETAIL_SHEET);
  const summary = ss.getSheetByName(SUMMARY_SHEET);
  const noteSh = ss.getSheetByName(NOTE_SHEET);

  const unitCatMap = {};
  if(detail){
    const values = detail.getDataRange().getValues();
    for(let i = 1; i < values.length; i++){
      const row = values[i];
      if(fmtDate_(row[0]) !== date) continue;
      const catName = row[1];
      if(!unitCatMap[catName]) unitCatMap[catName] = {name: catName, kind:'unit', subtypes: [], units: []};
      const subtype = (row[3] && row[3] !== '-') ? row[3] : '';
      const status = row[4] === 'พร้อมใช้' ? 'ready' : 'repair';
      unitCatMap[catName].units.push({id:'u'+i, number: row[2], subtype: subtype, status, remark: row[5] || ''});
      if(subtype && unitCatMap[catName].subtypes.indexOf(subtype) === -1) unitCatMap[catName].subtypes.push(subtype);
    }
  }

  const derivedLabels = {};
  Object.keys(unitCatMap).forEach(name => {
    const cat = unitCatMap[name];
    if(cat.subtypes.length){
      cat.subtypes.forEach(st => { derivedLabels[name + ' ' + st] = true; });
      if(cat.units.some(u => cat.subtypes.indexOf(u.subtype) === -1)) derivedLabels[name + ' อื่นๆ'] = true;
    } else {
      derivedLabels[name] = true;
    }
  });

  const categories = Object.keys(unitCatMap).map(k => unitCatMap[k]);

  if(summary){
    const values = summary.getDataRange().getValues();
    for(let i = 1; i < values.length; i++){
      const row = values[i];
      if(fmtDate_(row[0]) !== date) continue;
      const label = row[1];
      if(label === 'รวมทั้งหมด') continue;
      if(derivedLabels[label]) continue;
      categories.push({name: label, kind:'aggregate', total: Number(row[2]) || 0, repair: Number(row[4]) || 0, remark: row[5] || ''});
    }
  }

  let note = '';
  if(noteSh){
    const values = noteSh.getDataRange().getValues();
    for(let i = 1; i < values.length; i++){
      if(fmtDate_(values[i][0]) === date){ note = values[i][1] || ''; break; }
    }
  }
  return {categories, note};
}
